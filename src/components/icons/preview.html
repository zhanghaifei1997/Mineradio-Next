快速集成指南

方式一：直接导入 Vue 组件
<!-- 在任意 .vue 文件中 -->
<script setup lang="ts">
import { IconPlay, IconPause, IconSkipBack, IconSkipForward } from '@/components/icons'
</script>

<template>
  <!-- 替换之前的 emoji: '▶' → IconPlay -->
  <button @click="player.togglePlay()">
    <IconPlay v-if="!player.isPlaying" :size="24" />
    <IconPause v-else :size="24" />
  </button>

  <!-- 替换 emoji: '⏮' → IconSkipBack -->
  <button @click="player.prev()" title="上一首">
    <IconSkipBack :size="20" />
  </button>
</template>
方式二：使用统一 Icon 组件（按名称动态切换）
<script setup lang="ts">
import { Icon } from '@/components/icons/Icon.vue'
import { computed } from 'vue'

const playModeIcon = computed(() => {
  const map = {
    sequence: 'list-ordered',
    loop: 'repeat',
    single: 'repeat-1',
    shuffle: 'shuffle'
  }
  return map[player.playMode]
})
</script>

<template>
  <!-- 一个组件，自动切换图标 -->
  <Icon :name="playModeIcon" :size="18" />
</template>
Props 说明
// 所有图标组件共享相同的 Props 接口
interface IconProps {
  size?: number | string  // 默认: 20
}

// Icon 组件额外支持 name 属性
interface IconDynamicProps extends IconProps {
  name: 'play' | 'pause' | 'skip-back' | 'skip-forward' | 'volume' | 'volume-x'
      | 'shuffle' | 'repeat' | 'repeat-1' | 'list-ordered' | 'list'
      | 'settings' | 'search' | 'x' | 'minimize' | 'maximize'
      | 'chevron-down' | 'chevron-up' | 'chevron-left' | 'chevron-right'
      | 'music' | 'mic' | 'user' | 'clock' | 'folder' | 'equalizer'
      | 'keyboard' | 'info' | 'trash' | 'plus'
      | 'grip-horizontal' | 'grip-vertical' | 'download' | 'compact'
      | 'arrow-down' | 'layers' | 'box' | 'edit'
}
